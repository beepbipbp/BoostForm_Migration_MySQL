// Question Schema
const QuestionSchema = new Schema<QuestionInterface>({
  question_id: {
    type: Number,
    required: true,
  },
  page: {
    type: Number,
  },
  type: {
    type: String,
    required: true,
    enum: ["checkbox", "multiple", "paragraph"],
  },
  title: {
    type: String,
    default: "제목 없음",
  },
  option: {
    type: [String],
  },
  essential: {
    type: Boolean,
    default: false,
  },
  etc_added: {
    type: Boolean,
    default: false,
  },
});

// Form Schema
const FormSchema = new Schema<FormInterface>(
  {
    user_id: {
      type: Number,
      required: true,
    },
    title: {
      type: String,
      default: "제목 없음",
    },
    description: {
      type: String,
    },
    category: {
      type: String,
    },
    question_list: {
      type: [QuestionSchema],
    },
    accept_response: {
      type: Boolean,
      default: false,
    },
    on_board: {
      type: Boolean,
      default: false,
    },
    login_required: {
      type: Boolean,
      default: false,
    },
    response_modifiable: {
      type: Boolean,
      default: true,
    },
    response_count: {
      type: Number,
      default: 0,
    },
  },
  {
    timestamps: true,
  }
);

// Answer Schema
const AnswerSchema = new Schema<AnswerInterface>({
  question_id: {
    type: Number,
  },
  answer: {
    type: [String],
  },
});

// Response Schema
const ResponseSchema = new Schema<ResponseInterface>({
  user_id: {
    type: Number,
  },
  form_id: {
    type: String,
    index: true,
  },
  answer_list: {
    type: [AnswerSchema],
  },
});
